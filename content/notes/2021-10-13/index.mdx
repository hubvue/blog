AST（抽象语法树）是编译过程中最重要的一环，在获取到代码的AST后，可以对代码进行自动化分析以及自动化生成，Go将代码编译成AST可以使用`go/ast`这个库来实现。


```go
package main

import (
	"fmt"
	"go/ast"
	"go/parser"
	"go/token"
)

var src = `
	package main
	import "fmt"
	func main() {
		fmt.Println("Hello world")
	}
`

func main() {
	fileSet := token.NewFileSet()  //创建文件集，用于定位AST中节点的文件位置

	f, err := parser.ParseFile(fileSet, "", src, 0) //解析文件

	if err != nil {
		panic(err)
	}
	ast.Print(fileSet, f)  //输出AST
}
```

ast包也提供了遍历节点的方法
```go
package main

import (
	"fmt"
	"go/ast"
	"go/parser"
	"go/token"
)

var src = `
	package main
	import "fmt"
	func main() {
		fmt.Println("Hello world")
	}
`

func main() {
	fileSet := token.NewFileSet()  //创建文件集，用于定位AST中节点的文件位置

	f, err := parser.ParseFile(fileSet, "", src, 0) //解析文件

	if err != nil {
		panic(err)
	}
  // 遍历AST
	ast.Inspect(f, func(node ast.Node) bool {
		var s string
		switch t := node.(type) {
		case *ast.BasicLit:
			s = t.Value
		case *ast.Ident:
			s = t.Name
		}
		if s != "" {
			fmt.Printf("%s:\t%s\n", fileSet.Position(node.Pos()), s)
		}
		return true
	})
}
```
